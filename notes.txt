bikin laravel mix lokal:
1. composer create-project laravel/laravel
2. composer require laravel/ui
3. php artisan ui react
4. npm install && npm run dev

langkah :
1. composer update
2. php artisan key:generate
3. php artisan config:cache
4. npm install
5. npm run

server :
1. sudo service apache2 start
7. apt-get install libapache2-mod-php7.4
2. npm install - run production
3. composer install --optimize-autoloader --no-dev (sudo apt-get install php-xml kalo error)
4. php artisan optimize:clear
5. APP_ENV=production / php artisan key:generate
6. php artisan config:cache

troubleshoot:
1. keluarin dari folder
2. di etc/apache2/sites-available/000-default.conf -> ubah document root nya jadi public
